// ===== STORAGE KEY =====
const STORAGE_KEY = 'sdv-bundles-v1';

// ===== INIT =====
let completedItems = new Set();
let bundleCompleted = {};

function loadProgress() {
  try {
    const saved = localStorage.getItem(STORAGE_KEY);
    if (saved) {
      const data = JSON.parse(saved);
      completedItems = new Set(data.completedItems || []);
      bundleCompleted = data.bundleCompleted || {};
      return;
    }
  } catch(e) {}
  // Default: nothing completed
  completedItems = new Set();
  bundleCompleted = {};
}

function saveProgress() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify({
    completedItems: Array.from(completedItems),
    bundleCompleted: bundleCompleted
  }));
}

// ===== RENDER =====
function renderBundles() {
  const container = document.getElementById('bundle-rooms');
  let totalItems = 0;
  let completedCount = 0;
  let html = '';

  for (const [roomId, room] of Object.entries(BUNDLES)) {
    const bundles = room.bundles;
    let roomHtml = '';
    let roomTotal = 0;
    let roomDone = 0;

    for (const [bundleId, bundle] of Object.entries(bundles)) {
      let bTotal = 0;
      let bDone = 0;
      let itemsHtml = '';

      for (const [idx, item] of bundle.items.entries()) {
        const itemId = `${roomId}-${bundleId}-${idx}`;
        bTotal++;
        const completed = completedItems.has(itemId);
        if (completed) bDone++;
        itemsHtml += `
          <label class="bundle-item tier-${item.tier} ${completed ? 'completed' : ''}"
                 onclick="toggleItem('${roomId}','${bundleId}',${idx})">
            <span class="check-box ${completed ? 'checked' : ''}">${completed ? '✓' : '○'}</span>
            <span class="item-name">${item.name}</span>
            <span class="item-needed">×${item.needed}</span>
          </label>`;
      }

      roomTotal += bTotal;
      roomDone += bDone;
      const bundleComplete = bDone >= bTotal;
      bundleCompleted[bundleId] = bundleComplete;

      itemsHtml += `<div class="bundle-reward">🎁 Reward: ${bundle.reward}</div>`;

      itemsHtml = `<h4 class="bundle-title ${bundleComplete ? 'bundle-done' : ''}">
        <span class="bundle-check">${bundleComplete ? '✅' : '⬜'}</span>
        ${bundle.name}
        <span class="bundle-count">${bDone}/${bTotal}</span>
      </h4>` + itemsHtml;

      roomHtml += itemsHtml;
    }

    totalItems += roomTotal;
    completedCount += roomDone;
    const roomComplete = roomDone >= roomTotal;

    html += `
      <div class="bundle-room" style="border-color: ${room.color}">
        <div class="room-header" onclick="toggleRoom('${roomId}')">
          <h3 class="${roomComplete ? 'room-done' : ''}">
            ${roomComplete ? '🌟' : '📂'} ${room.name}
            <span class="room-count">${roomDone}/${roomTotal}</span>
          </h3>
        </div>
        <div class="room-content" id="room-${roomId}">
          ${roomHtml}
        </div>
      </div>`;
  }

  container.innerHTML = html;

  // Update stats
  document.getElementById('total-items').textContent = totalItems;
  document.getElementById('completed-items').textContent = completedCount;
  const pct = totalItems > 0 ? Math.round(completedCount / totalItems * 100) : 0;
  document.getElementById('progress-pct').textContent = `${pct}%`;
  document.getElementById('progress-bar').style.width = `${pct}%`;

  saveProgress();
}

function toggleItem(roomId, bundleId, idx) {
  const itemId = `${roomId}-${bundleId}-${idx}`;
  if (completedItems.has(itemId)) {
    completedItems.delete(itemId);
  } else {
    completedItems.add(itemId);
  }
  renderBundles();
}

function toggleRoom(roomId) {
  const content = document.getElementById(`room-${roomId}`);
  content.style.display = content.style.display === 'none' ? 'block' : 'none';
}

function resetProgress() {
  if (confirm('⚠️ Reset ALL bundle progress? This cannot be undone.')) {
    if (confirm('Are you sure? All items will be unchecked.')) {
      completedItems = new Set();
      bundleCompleted = {};
      saveProgress();
      renderBundles();
    }
  }
}

function exportProgress() {
  const data = {
    exported: new Date().toISOString(),
    game: 'Stardew Valley 1.6',
    completedItems: Array.from(completedItems),
    stats: {
      total: document.getElementById('total-items').textContent,
      completed: document.getElementById('completed-items').textContent,
      progress: document.getElementById('progress-pct').textContent
    }
  };
  const text = JSON.stringify(data, null, 2);
  navigator.clipboard.writeText(text).then(() => {
    alert('✅ Progress copied to clipboard! Paste it somewhere safe.');
  }).catch(() => {
    prompt('📋 Copy this text:', text);
  });
}

// ===== START =====
loadProgress();
renderBundles();
</script>
